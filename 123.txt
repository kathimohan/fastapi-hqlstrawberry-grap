@event.listens_for(Item, "before_delete")
def remember_latest_parent(mapper, connection, target):
    session = Session.object_session(target)
    if session and "latest" in _split_labels(target.labels):
        session.info.setdefault("deleted_latest_parents", set()).add(target.parent_id)

@event.listens_for(Session, "after_flush")
def reassign_latest_after_delete(session, flush_context):
    parent_ids = session.info.pop("deleted_latest_parents", set())
    for pid in parent_ids:
        sibling = (
            session.query(Item)
            .filter(Item.parent_id == pid)
            .order_by(Item.version.desc())
            .first()
        )
        if sibling:
            labels = _split_labels(sibling.labels)
            if "latest" not in labels:
                labels.append("latest")
                sibling.labels = _join_labels(labels)
