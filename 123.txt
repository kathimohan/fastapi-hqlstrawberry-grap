def build_conditions(model, arg_map):
    """
    model: SQLAlchemy model
    arg_map: dict mapping {arg_value: model_column}
    """
    conditions = []
    for value, column in arg_map.items():
        if value is not None:
            conditions.append(column == value)
    return conditions

def get_filtered_query(session, a=None, b=None, c=None, d=None):
    arg_map = {
        a: Model.col_a,
        b: Model.col_b,
        c: Model.col_c,
        d: Model.col_d,
    }

    conditions = build_conditions(Model, arg_map)
    query = session.query(Model)

    if conditions:
        query = query.filter(and_(*conditions))

    return query.all()
