from typing import Any, Dict, Tuple

def build_params_with_msg(**kwargs: Any) -> Tuple[Dict[str, Any], str]:
    """Build params dict excluding None values, and return a message string.

    Args:
        **kwargs: Arbitrary keyword arguments.

    Returns:
        Tuple[Dict[str, Any], str]:
            - Dict containing only keys whose values are not None.
            - String message listing received args (e.g. "the args received are a=5, b=6").

    Notes:
        - Mandatory args should be validated before calling this function.
    """
    params = {k: v for k, v in kwargs.items() if v is not None}
    msg = (
        "the args received are " + ", ".join(f"{k}={v}" for k, v in params.items())
        if params else "no args received"
    )
    return params, msg
