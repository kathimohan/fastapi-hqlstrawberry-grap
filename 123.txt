import pandas as pd

df = pd.DataFrame({
    "id": [1, 2, 3],
    "details": [
        [{"a": 10, "b": 20}],   # list of dicts
        None,                   # no dicts
        [{"a": 30, "c": 40}]    # different keys
    ]
})

def expand_list_of_dicts(row):
    if isinstance(row, list) and all(isinstance(d, dict) for d in row):
        # merge dicts inside list into one
        merged = {}
        for d in row:
            merged.update(d)
        return merged
    return {}

# Expand details
expanded = df["details"].apply(expand_list_of_dicts)

# Normalize into columns and join back
df_expanded = pd.concat([df.drop(columns=["details"]),
                         pd.json_normalize(expanded)], axis=1)

print(df_expanded)
