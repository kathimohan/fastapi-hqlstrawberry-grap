from sqlalchemy import Column, Integer, String, ForeignKey
from sqlalchemy.orm import relationship, declarative_base

Base = declarative_base()

class Parent(Base):
    __tablename__ = "parents"
    id = Column(Integer, primary_key=True)
    parent_id = Column(String)
    name = Column(String)

    children = relationship("Child", back_populates="parent")

class Child(Base):
    __tablename__ = "children"
    id = Column(Integer, primary_key=True)
    parent_id = Column(Integer, ForeignKey("parents.id"))
    name = Column(String)

    parent = relationship("Parent", back_populates="children")

from sqlalchemy import and_
from sqlalchemy.sql.sqltypes import String, Text, Date, Integer, Float
from sqlalchemy.dialects.postgresql import JSONB

OPS = {
    "eq": lambda c, v: c == v,
    "neq": lambda c, v: c != v,
    "gt": lambda c, v: c > v,
    "lt": lambda c, v: c < v,
    "gte": lambda c, v: c >= v,
    "lte": lambda c, v: c <= v,
    "contains": lambda c, v: c.ilike(f"%{v}%"),
    "startswith": lambda c, v: c.ilike(f"{v}%"),
    "endswith": lambda c, v: c.ilike(f"%{v}"),
    "exact": lambda c, v: c == v,
    "iexact": lambda c, v: c.ilike(v),
}

def build_filter(model_alias_map, search_conditions):
    filters = []
    for cond in search_conditions:
        for field, ops_list in cond.items():
            # Expect format: "parent.name" or "child.name"
            model_name, col_name = field.split(".", 1)
            model_alias = model_alias_map[model_name]
            column = getattr(model_alias, col_name)

            for op_dict in ops_list:
                op_type = op_dict["type"]
                value = op_dict["searchterm"]
                if op_type in OPS:
                    filters.append(OPS[op_type](column, value))
    return and_(*filters)

def apply_order_by(query, model_alias_map, order_by_list):
    if not order_by_list:
        return query
    for order_item in order_by_list:
        field = order_item["field"]   # e.g., "child.name"
        direction = order_item.get("direction", "asc")
        model_name, col_name = field.split(".", 1)
        column = getattr(model_alias_map[model_name], col_name)
        if direction.lower() == "desc":
            query = query.order_by(column.desc())
        else:
            query = query.order_by(column.asc())
    return query

def apply_limit_offset(query, limit, offset, max_limit=500, default_limit=500):
    applied_limit = min(limit or default_limit, max_limit)
    applied_offset = offset or 0
    return query.limit(applied_limit).offset(applied_offset), applied_limit, applied_offset


from sqlalchemy.orm import aliased

class SearchService:
    def __init__(self, session_factory):
        self.session_factory = session_factory

    def search_parent_child(self, search_conditions, return_fields, order_by=None, limit=None, offset=None):
        with self.session_factory() as db:
            ParentAlias = aliased(Parent)
            ChildAlias = aliased(Child)

            model_alias_map = {
                "parent": ParentAlias,
                "child": ChildAlias
            }

            query = db.query(ParentAlias, ChildAlias).join(ChildAlias, ParentAlias.id == ChildAlias.parent_id)

            # Apply filters
            if search_conditions:
                filter_expr = build_filter(model_alias_map, search_conditions)
                query = query.filter(filter_expr)

            # Select only required fields
            entities = []
            for field in return_fields:
                model_name, col_name = field.split(".", 1)
                model_alias = model_alias_map[model_name]
                entities.append(getattr(model_alias, col_name).label(f"{model_name}_{col_name}"))

            if entities:
                query = db.query(*entities).join(ChildAlias, ParentAlias.id == ChildAlias.parent_id)

            # Apply ordering
            query = apply_order_by(query, model_alias_map, order_by)

            # Pagination
            query, applied_limit, applied_offset = apply_limit_offset(query, limit, offset)

            rows = query.all()
            results = [dict(r._mapping) for r in rows]

            return {
                "results": results,
                "total_count": len(results),
                "limit": applied_limit,
                "offset": applied_offset
            }

from typing import List, Optional, Union
from sqlalchemy.inspection import inspect

def model_to_dict(
    instance: Union[object, List[object]], 
    fields: Optional[List[str]] = None
) -> Union[dict, List[dict]]:
    """
    Convert a SQLAlchemy ORM instance or list of instances to a dict (or list of dicts).
    Optionally, filter to specific fields.
    """
    def _to_dict(obj):
        column_attrs = inspect(obj).mapper.column_attrs
        all_data = {c.key: getattr(obj, c.key) for c in column_attrs}
        if fields:
            return {k: all_data[k] for k in fields if k in all_data}
        return all_data

    if isinstance(instance, list):
        return [_to_dict(obj) for obj in instance]
    return _to_dict(instance)
